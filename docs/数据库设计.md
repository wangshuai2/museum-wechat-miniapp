## 集合设计

### 1. users (用户集合)

**用途**: 存储用户基本信息

**Schema**:
```typescript
{
  _id: ObjectId,
  username: String,           // 用户名，唯一
  email: String,              // 邮箱，唯一
  password: String,           // 加密存储
  phone: String,              // 手机号
  avatar: String,             // 头像 URL
  role: String,               // 角色：user, admin, super_admin
  status: String,             // 状态：active, inactive, banned
  profile: {
    nickname: String,
    gender: String,           // male, female, other
    birthday: Date,
    address: String
  },
  preferences: {
    language: String,         // zh-CN, en-US
    notifications: Boolean
  },
  lastLoginAt: Date,
  createdAt: Date,
  updatedAt: Date
}
```

**索引**:
```javascript
{ username: 1, unique: true }
{ email: 1, unique: true }
{ phone: 1, unique: true, sparse: true }
{ role: 1 }
{ status: 1 }
```

### 2. exhibits (展品集合)

**用途**: 存储展品信息

**Schema**:
```typescript
{
  _id: ObjectId,
  name: String,               // 展品名称
  description: String,        // 简介
  detailedDescription: String, // 详细描述
  images: [String],           // 图片 URL 数组
  coverImage: String,         // 封面图
  location: String,           // 位置
  floor: String,              // 楼层
  hall: String,               // 展厅
  category: String,           // 分类
  tags: [String],             // 标签
  audioGuide: String,         // 语音导览 URL
  videoUrl: String,           // 视频 URL
  historicalBackground: String, // 历史背景
  culturalValue: String,      // 文化价值
  dimensions: {               // 尺寸
    height: Number,
    width: Number,
    depth: Number,
    unit: String              // cm, m
  },
  material: String,           // 材质
  period: String,             // 年代
  artist: String,             // 作者/艺术家
  acquisitionInfo: String,    // 收藏信息
  isActive: Boolean,          // 是否展示
  viewCount: Number,          // 浏览次数
  favoriteCount: Number,      // 收藏次数
  rating: {                   // 评分
    average: Number,
    count: Number
  },
  relatedExhibits: [ObjectId], // 相关展品
  createdAt: Date,
  updatedAt: Date
}
```

**索引**:
```javascript
{ name: 'text', description: 'text' }  // 全文索引
{ category: 1 }
{ location: 1 }
{ tags: 1 }
{ isActive: 1 }
{ viewCount: -1 }
{ 'rating.average': -1 }
{ floor: 1, hall: 1 }
```

### 3. reservations (预约集合)

**用途**: 存储预约记录

**Schema**:
```typescript
{
  _id: ObjectId,
  userId: ObjectId,           // 用户 ID
  exhibitId: ObjectId,        // 展品 ID (可选，为空表示场馆预约)
  date: Date,                 // 预约日期
  timeSlot: String,           // 时段：09:00-10:00
  visitorCount: Number,       // 参观人数
  contactName: String,        // 联系人
  contactPhone: String,       // 联系电话
  contactEmail: String,       // 联系邮箱
  status: String,             // pending, confirmed, cancelled, completed, no_show
  qrCode: String,             // 二维码 URL
  remark: String,             // 备注
  cancelReason: String,       // 取消原因
  checkedInAt: Date,          // 签到时间
  checkedOutAt: Date,         // 签出时间
  reminderSent: Boolean,      // 是否已发送提醒
  createdAt: Date,
  updatedAt: Date
}
```

**索引**:
```javascript
{ userId: 1, createdAt: -1 }
{ exhibitId: 1, date: 1 }
{ date: 1, timeSlot: 1 }
{ status: 1 }
{ userId: 1, status: 1, date: -1 }
```

### 4. visits (访问记录集合)

**用途**: 存储用户访问记录

**Schema**:
```typescript
{
  _id: ObjectId,
  userId: ObjectId,           // 用户 ID
  exhibitId: ObjectId,        // 展品 ID
  action: String,             // 动作：view, like, favorite, share
  duration: Number,           // 停留时长 (秒)
  device: String,             // 设备类型：wechat, flutter, web
  ip: String,                 // IP 地址
  userAgent: String,          // 用户代理
  metadata: {                 // 额外信息
    referer: String,
    searchKeyword: String
  },
  createdAt: Date
}
```

**索引**:
```javascript
{ userId: 1, createdAt: -1 }
{ exhibitId: 1, createdAt: -1 }
{ userId: 1, exhibitId: 1, action: 1 }
{ createdAt: -1 }             // 用于统计分析
```

### 6. reviews (评价集合)

**用途**: 存储用户评价

**Schema**:
```typescript
{
  _id: ObjectId,
  userId: ObjectId,
  exhibitId: ObjectId,
  reservationId: ObjectId,    // 关联预约 (可选)
  rating: Number,             // 评分 1-5

  status: String,             // pending, approved, rejected
  likes: Number,              // 点赞数
  replies: [{                 // 回复
    userId: ObjectId,
    content: String,
    createdAt: Date
  }],
  createdAt: Date,
  updatedAt: Date
}
```

**索引**:
```javascript
{ exhibitId: 1, rating: -1 }
{ userId: 1, createdAt: -1 }
{ status: 1, createdAt: -1 }
```

## 性能优化建议

### 1. 查询优化
- 使用投影只获取需要的字段
- 合理使用索引
- 避免全表扫描
- 使用聚合管道处理复杂查询

### 2. 写入优化
- 批量写入
- 合理使用 upsert
- 避免频繁更新大文档

### 3. 存储优化
- 大文件使用 GridFS
- 历史数据归档
- 定期清理过期数据

### 4. 监控指标
- 慢查询日志
- 连接池使用率
- 存储空间使用
- 索引命中率
