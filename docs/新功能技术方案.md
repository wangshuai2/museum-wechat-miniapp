## 一、数据模型设计

### 1.1 Notes 集合（笔记）

```typescript
// src/backend/src/notes/schemas/note.schema.ts
import { Prop, Schema, SchemaFactory } from '@nestjs/mongoose';
import { Document, Types } from 'mongoose';

export type NoteDocument = Note & Document;

@Schema({ timestamps: true })
export class Note {
  // === 关联字段 ===
  @Prop({ type: Types.ObjectId, ref: 'User', required: true, index: true })
  userId: Types.ObjectId;

  @Prop({ type: Types.ObjectId, ref: 'Museum', required: true, index: true })
  museumId: Types.ObjectId;

  @Prop({ type: Types.ObjectId, ref: 'Exhibit', index: true })
  exhibitId?: Types.ObjectId;

  @Prop({ type: Types.ObjectId, ref: 'Footprint' })
  footprintId?: Types.ObjectId; // 关联参观记录（可选）

  // === 内容字段 ===
  @Prop({ trim: true, maxlength: 100 })
  title?: string; // 笔记标题（可选）

  @Prop({ required: true, maxlength: 5000 })
  content: string; // 笔记内容（纯文本，限制5000字）

  @Prop({ type: [String], default: [] })
  images: string[]; // 图片 URL 数组（最多9张）

  @Prop({ type: [String], default: [] })
  tags: string[]; // 标签数组

  // === 状态字段 ===
  @Prop({ enum: ['private', 'public'], default: 'private' })
  visibility: string; // 可见性

  @Prop({ type: Number, default: 0, min: 0 })
  likeCount: number; // 点赞数（公开笔记）

  @Prop({ type: Number, default: 0, min: 0 })
  viewCount: number; // 浏览数

  // === 冗余字段（优化查询） ===
  @Prop({ type: String })
  museumName?: string; // 博物馆名称（冗余）

  @Prop({ type: String })
  exhibitName?: string; // 展品名称（冗余）

  @Prop({ type: String })
  museumImage?: string; // 博物馆封面（冗余）

  @Prop({ type: String })
  exhibitImage?: string; // 展品图片（冗余）
}

export const NoteSchema = SchemaFactory.createForClass(Note);

// 索引设计
NoteSchema.index({ userId: 1, createdAt: -1 }); // 用户笔记列表查询
NoteSchema.index({ museumId: 1, createdAt: -1 }); // 博物馆笔记列表查询
NoteSchema.index({ exhibitId: 1, createdAt: -1 }, { sparse: true }); // 展品笔记列表查询
NoteSchema.index({ visibility: 1, createdAt: -1 }); // 公开笔记列表查询
NoteSchema.index({ userId: 1, museumId: 1 }); // 检查用户在某博物馆是否有笔记
NoteSchema.index({ userId: 1, exhibitId: 1 }, { sparse: true }); // 检查用户在某展品是否有笔记
NoteSchema.index({ tags: 1 }); // 标签查询
NoteSchema.index({ content: 'text', title: 'text' }); // 全文搜索
```

### 1.2 Favorites 集合（收藏）

```typescript
// src/backend/src/favorites/schemas/favorite.schema.ts
import { Prop, Schema, SchemaFactory } from '@nestjs/mongoose';
import { Document, Types } from 'mongoose';

export type FavoriteDocument = Favorite & Document;

@Schema({ timestamps: true })
export class Favorite {
  // === 关联字段 ===
  @Prop({ type: Types.ObjectId, ref: 'User', required: true, index: true })
  userId: Types.ObjectId;

  @Prop({ 
    type: String, 
    required: true, 
    enum: ['museum', 'exhibit', 'exhibition'],
    index: true
  })
  targetType: string; // 收藏目标类型

  @Prop({ type: Types.ObjectId, required: true, refPath: 'targetType' })
  targetId: Types.ObjectId; // 收藏目标 ID

  // === 状态字段 ===
  @Prop({ 
    type: String, 
    enum: ['want', 'visited'], 
    default: 'want' 
  })
  status: string; // 想去/已去过（仅博物馆收藏有效）

  @Prop({ type: String, maxlength: 200 })
  note?: string; // 收藏备注

  @Prop({ type: Number, default: 0 })
  sortOrder: number; // 排序顺序

  // === 冗余字段（优化查询） ===
  @Prop({ type: String, required: true })
  targetName: string; // 目标名称

  @Prop({ type: String })
  targetImage?: string; // 目标图片

  @Prop({ type: String })
  targetAddress?: string; // 目标地址（博物馆地址）

  @Prop({ type: [String], default: [] })
  targetTags?: string[]; // 目标标签
}

export const FavoriteSchema = SchemaFactory.createForClass(Favorite);

// 唯一索引：同一用户不能重复收藏同一目标
FavoriteSchema.index({ userId: 1, targetType: 1, targetId: 1 }, { unique: true });

// 查询索引
FavoriteSchema.index({ userId: 1, createdAt: -1 }); // 用户收藏列表
FavoriteSchema.index({ userId: 1, targetType: 1, status: 1 }); // 按类型和状态筛选
FavoriteSchema.index({ userId: 1, sortOrder: 1 }); // 排序查询
FavoriteSchema.index({ targetId: 1 }); // 统计收藏数量
```

Authorization: Bearer <token>

## 五、技术风险与应对

| 风险 | 影响 | 概率 | 应对措施 |
|------|------|------|----------|
| 图片存储成本 | 成本压力 | 中 | 限制图片数量和大小，使用 CDN |
| 笔记内容审核 | 平台风险 | 中 | 敏感词过滤 + 人工审核机制 |
| 收藏数据量增长 | 性能问题 | 低 | 分页查询 + 索引优化 |
| 并发收藏操作 | 数据一致性 | 低 | 使用 MongoDB 原子操作 |
