## 三、技术实现方案

### 3.1 地理位置查询

```typescript
// 附近博物馆查询
async findNearby(lng: number, lat: number, maxDistance: number = 10000) {
  return this.museumModel.find({
    location: {
      $near: {
        $geometry: {
          type: 'Point',
          coordinates: [lng, lat]
        },
        $maxDistance: maxDistance  // 米
      }
    },
    status: 'open'
  });
}

// 范围查询
async findWithinBounds(southwest: [number, number], northeast: [number, number]) {
  return this.museumModel.find({
    location: {
      $geoWithin: {
        $box: [southwest, northeast]
      }
    }
  });
}
```

### 3.2 全文搜索

```typescript
// 搜索博物馆
async searchMuseums(keyword: string, filters: SearchFilters) {
  const query: any = {
    $text: { $search: keyword },
    status: 'open'
  };
  
  if (filters.city) query.city = filters.city;
  if (filters.type) query.type = filters.type;
  
  return this.museumModel
    .find(query, { score: { $meta: 'textScore' } })
    .sort({ score: { $meta: 'textScore' } })
    .limit(20);
}
```

### 3.3 统计聚合

```typescript
// 用户足迹统计
async getUserStatistics(userId: string) {
  const result = await this.footprintModel.aggregate([
    { $match: { userId: new ObjectId(userId) } },
    {
      $facet: {
        overview: [
          {
            $group: {
              _id: null,
              totalVisits: { $sum: 1 },
              totalMuseums: { $addToSet: '$museumId' },
              totalCities: { $addToSet: '$city' }
            }
          }
        ],
        byType: [
          {
            $lookup: {
              from: 'museums',
              localField: 'museumId',
              foreignField: '_id',
              as: 'museum'
            }
          },
          { $unwind: '$museum' },
          {
            $group: {
              _id: '$museum.type',
              count: { $sum: 1 }
            }
          }
        ],
        monthly: [
          {
            $group: {
              _id: { $dateToString: { format: '%Y-%m', date: '$visitDate' } },
              count: { $sum: 1 }
            }
          },
          { $sort: { _id: -1 } },
          { $limit: 12 }
        ]
      }
    }
  ]);
  
  return result[0];
}
```

### 4.1 博物馆数据导入

```typescript
// 数据导入脚本
async importMuseums(data: MuseumData[]) {
  const bulkOps = data.map(item => ({
    updateOne: {
      filter: { name: item.name },
      update: { $set: item },
      upsert: true
    }
  }));
  
  await this.museumModel.bulkWrite(bulkOps);
}
```
