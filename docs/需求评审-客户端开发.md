### 3.2 离线同步机制

```dart
class SyncService {
  final HiveBox<FootprintLocalModel> localBox;
  final ApiService apiService;
  
  // 同步离线数据
  Future<void> syncPendingFootprints() async {
    final pendingItems = localBox.values
        .where((item) => !item.isSynced)
        .toList();
    
    for (final item in pendingItems) {
      try {
        await apiService.createFootprint(item.toDto());
        item.isSynced = true;
        await item.save();
      } catch (e) {
        // 记录失败，下次重试
        log('Sync failed: $e');
      }
    }
  }
  
  // 监听网络状态
  void listenNetworkStatus() {
    Connectivity().onConnectivityChanged.listen((status) {
      if (status != ConnectivityResult.none) {
        syncPendingFootprints();
      }
    });
  }
}
```

### 3.3 缓存策略

```dart
class CacheService {
  // 博物馆数据缓存（7天有效）
  Future<List<Museum>> getMuseums({bool forceRefresh = false}) async {
    final cacheKey = 'museums_list';
    final cached = await _getCached<Museum>(cacheKey);
    
    if (!forceRefresh && cached != null && _isValid(cached.timestamp, days: 7)) {
      return cached.data;
    }
    
    final fresh = await apiService.getMuseums();
    await _setCache(cacheKey, fresh);
    return fresh;
  }
  
  // 图片缓存
  Widget buildCachedImage(String url) {
    return CachedNetworkImage(
      imageUrl: url,
      placeholder: (context, url) => ShimmerLoading(),
      errorWidget: (context, url, error) => Icon(Icons.error),
      memCacheWidth: 800,  // 限制内存缓存大小
    );
  }
}
```

```dart
class QRScanService {
  // 扫描博物馆二维码
  Future<Museum> scanMuseumQR() async {
    final result = await qrCodeScanner.scan();
    
    if (result != null && result.startsWith('museum://')) {
      final museumId = result.replaceAll('museum://', '');
      return apiService.getMuseum(museumId);
    }
    
    throw InvalidQRError('无效的博物馆二维码');
  }
  
  // 签到
  Future<Footprint> checkIn(Museum museum, Location location) async {
    final footprint = FootprintDto(
      museumId: museum.id,
      visitDate: DateTime.now(),
      location: location,
    );
    
    return apiService.createFootprint(footprint);
  }
}
```

class ExportService {
  // 导出为图片
  Future<File> exportAsImage(List<Footprint> footprints) async {
    final recorder = PictureRecorder();
    final canvas = Canvas(recorder);
    
    // 绘制足迹统计图
    _drawStatistics(canvas, footprints);
    
    final picture = recorder.endRecording();
    final image = await picture.toImage(1080, 1920);
    final byteData = await image.toByteData(format: ImageByteFormat.png);
    
    final file = File('${savePath}/footprint_${DateTime.now().millisecondsSinceEpoch}.png');
    await file.writeAsBytes(byteData.buffer.asUint8List());
    
    return file;
  }
