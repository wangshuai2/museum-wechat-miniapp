## 🔍 常见问题模式

### 1. 空值处理
```typescript
// ❌ 不良实践
const name = user.name.toUpperCase();

// ✅ 推荐做法
const name = user?.name?.toUpperCase() ?? '默认名称';
```

### 2. 异步错误处理
```typescript
// ❌ 不良实践
async function getData() {
  const res = await fetch('/api');
  return res.json();
}

// ✅ 推荐做法
async function getData() {
  try {
    const res = await fetch('/api');
    if (!res.ok) throw new Error('请求失败');
    return await res.json();
  } catch (error) {
    logger.error('获取数据失败', error);
    throw error;
  }
}
```

### 3. 数据库查询优化
```typescript
// ❌ 不良实践 - N+1 问题
const users = await userModel.find();
for (const user of users) {
  user.profile = await profileModel.findOne({ userId: user.id });
}

// ✅ 推荐做法 - 聚合查询
const users = await userModel.aggregate([
  {
    $lookup: {
      from: 'profiles',
      localField: '_id',
      foreignField: 'userId',
      as: 'profile',
    },
  },
]);
```

### 4. 内存泄漏预防
```dart
// ❌ 不良实践
class MyPage extends StatefulWidget {
  @override
  _MyPageState createState() => _MyPageState();
}

class _MyPageState extends State<MyPage> {
  Timer? _timer;
  
  @override
  void initState() {
    super.initState();
    _timer = Timer.periodic(Duration(seconds: 1), (t) {
      // 定时任务
    });
  }
  // 缺少 dispose
}

// ✅ 推荐做法
class _MyPageState extends State<MyPage> {
  Timer? _timer;
  
  @override
  void initState() {
    super.initState();
    _timer = Timer.periodic(Duration(seconds: 1), (t) {
      // 定时任务
    });
  }
  
  @override
  void dispose() {
    _timer?.cancel();
    super.dispose();
  }
}
```
